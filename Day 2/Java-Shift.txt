//Example Of Shift Operators :

class ShiftOperators {
    public static void main(String[] args)
    {
  
        int a = 5;
        int b = -10;
  
        // left shift operator
       // similar to 5*(2^2) 
        System.out.println("a<<2 = " + (a << 2));
  
        // right shift operator
        // similar to 5/(2^2) 
        System.out.println("b>>2 = " + (b >> 2));
      }
}
